% Kompiler med: pdflatex ER-diagram.tex
% Rekompiler alltid etter endringer i diagrammet.
%
% ER-diagram med tikz-er2: entity, weak entity, relationship, ident relationship,
% \key{} for nøkler, \dashuline{} for partiell nøkkel (ulem, lastet av tikz-er2).
% Entitetsnavn: \entnavn{...} (fet, oppreist) for å skille fra attributter (kursiv).
\documentclass[border=10pt]{standalone}
\usepackage{xcolor}
\usepackage{tikz-er2}
\usetikzlibrary{positioning,arrows.meta}

\definecolor{entityblue}{RGB}{173, 216, 230}
\newcommand{\entnavn}[1]{\normalfont\textbf{#1}}

\begin{document}
\begin{tikzpicture}[
  >={Stealth[length=3mm, width=2.2mm]},
  every entity/.style={
    fill=entityblue,
    minimum width=2.5cm,
    minimum height=2cm,
    align=left,
    font=\itshape,
    inner sep=4pt,
  },
  every weak entity/.style={
    fill=entityblue,
    minimum width=2.5cm,
    minimum height=2cm,
    align=left,
    font=\itshape,
    inner sep=4pt,
  },
  every relationship/.style={
    fill=entityblue,
    aspect=2,
    minimum width=1.4cm,
    minimum height=0.9cm,
    inner sep=2pt,
  },
  every ident relationship/.style={
    fill=entityblue,
    aspect=2,
    minimum width=1.4cm,
    minimum height=0.9cm,
    inner sep=2pt,
  }
]

  % ========== SONE 1: Senter og Fasilitet (øverst) ==========
  \node[entity] (senter) at (0, 26) {\entnavn{Senter}\\
    \key{ID}\\
    navn\\
    adresse\\
    \hspace{1em} gate\\
    \hspace{1em} nummer\\
    åpningstid\\
    stengetid\\
    bemannet\_fra\\
    bemannet\_til};
  \node[relationship] (senter-fasilitet) at (7.5, 26) {\normalsize har};
  \node[entity] (fasilitet) at (15, 26) {\entnavn{Fasilitet}\\
    \key{ID}\\
    type\\
    beskrivelse};

  % ========== SONE 2: Senter → Sal → Tredemølle / Sykkel (venstre vertikal kjede) ==========
  \node[ident relationship] (senter-sal) at (0, 19) {\normalsize har\_sal};
  \node[weak entity] (sal) at (0, 11) {\entnavn{Sal}\\
    \dashuline{ID}\\
    type\\
    kapasitet};
  \node[ident relationship] (sal-tredemolle) at (-7.5, 11) {\normalsize har\_tredemølle};
  \node[weak entity] (tredemolle) at (-7.5, 4) {\entnavn{Tredemølle}\\
    \dashuline{nummer}\\
    produsent\\
    maksimal\_hastighet\\
    maksimal\_stigning};
  \node[ident relationship] (sal-sykkel) at (0, 5.5) {\normalsize har\_sykkel};
  \node[weak entity] (sykkel) at (0, 0) {\entnavn{Sykkel}\\
    \dashuline{nummer}\\
    bluetooth};

  % ========== SONE 3: Idrettslagstime (sentrum) – booket_for fra Sal ==========
  \node[entity] (idrettslagstime) at (9.5, 7.5) {\entnavn{Idrettslagstime}\\
    \key{ID}\\
    tidsrom\\
    \hspace{1em} start\\
    \hspace{1em} slutt\\
    beskrivelse};
  \node[relationship] (booket-for-sal) at (4.5, 9) {\normalsize booket\_for};
  \node[relationship] (booket-for-gruppe) at (5, 3) {\normalsize booket\_for};
  \node[relationship] (arrangerer) at (15, 7.5) {\normalsize arrangerer};
  \node[entity] (idrett) at (21, 7.5) {\entnavn{Idrettslag}\\
    \key{ID}\\
    navn\\
    beskrivelse};

  % ========== SONE 4: Gruppeaktivitet og Brukerprofil ==========
  \node[entity] (gruppe) at (9.5, -5.5) {\entnavn{Gruppeaktivitet}\\
    \key{ID}\\
    type\\
    tidsrom\\
    \hspace{1em} start\\
    \hspace{1em} slutt\\
    beskrivelse};
  \node[relationship] (instruktor-for) at (14, -2.5) {\normalsize instruktør\_for};
  \node[relationship] (moter-til) at (14, -5.5) {\normalsize møter\_til};
  \node[relationship] (pameldt-til) at (14, -8.5) {\normalsize påmeldt\_til};
  \node[entity] (bruker) at (21, -5.5) {\entnavn{Brukerprofil}\\
    \key{ID}\\
    navn\\
    epost\\
    telefon};

  % ========== SONE 5: Brukerprofil → Økt og Prikk ==========
  \node[ident relationship] (bruker-okt) at (25, -5.5) {\normalsize har\_økt};
  \node[weak entity] (okt) at (30, -5.5) {\entnavn{Økt}\\
    \dashuline{ID}\\
    innsjekk\\
    utsjekk};
  \node[ident relationship] (bruker-prikk) at (25, -11) {\normalsize har\_prikk};
  \node[weak entity] (prikk) at (30, -11) {\entnavn{Prikk}\\
    \dashuline{dato\_fikk\_prikk}};

  % ========== Kanter ==========
  % Senter -- har -- Fasilitet
  \draw (senter.east) -- (senter-fasilitet.west);
  \draw (fasilitet.west) -- (senter-fasilitet.east);

  % Senter -- har_sal -- Sal
  \draw[->] (senter-sal.north) -- (senter.south);
  \draw[double, double distance=1.5pt] (senter-sal.south) -- (sal.north);

  % Sal -- har_tredemølle -- Tredemølle
  \draw[->] (sal-tredemolle.east) -- (sal.west);
  \draw[double, double distance=1.5pt] (sal-tredemolle.south) -- (tredemolle.north);

  % Sal -- har_sykkel -- Sykkel
  \draw[->] (sal-sykkel.north) -- (sal.south);
  \draw[double, double distance=1.5pt] (sal-sykkel.south) -- (sykkel.north);

  % Sal -- booket_for -- Idrettslagstime
  \draw[->] (booket-for-sal.west) -- (sal.east);
  \draw[double, double distance=1.5pt] (idrettslagstime.west) -- (booket-for-sal.east);

  % Sal -- booket_for -- Gruppeaktivitet
  \draw[->] (booket-for-gruppe.west) -- (sal.east);
  \draw[double, double distance=1.5pt] (gruppe.north west) -- (booket-for-gruppe.south east);

  % Idrettslagstime -- arrangerer -- Idrettslag
  \draw[double, double distance=1.5pt] (idrettslagstime.east) -- (arrangerer.west);
  \draw[->] (arrangerer.east) -- (idrett.west);

  % Idrettslag -- er_medlem -- Brukerprofil
  \node[relationship] (er-medlem) at (21, 0.5) {\normalsize er\_medlem};
  \draw (bruker.north) -- (er-medlem.south);
  \draw (idrett.south) -- (er-medlem.north);

  % Gruppeaktivitet -- instruktør_for / møter_til / påmeldt_til -- Brukerprofil (anker slik at linjene ikke krysser)
  \draw[double, double distance=1.5pt] (gruppe.north east) -- (instruktor-for.south west);
  \draw[->] (instruktor-for.south east) -- (bruker.north west);
  \draw (gruppe.east) -- (moter-til.west);
  \draw (moter-til.east) -- (bruker.west);
  \draw (gruppe.south east) -- (pameldt-til.north west);
  \draw (bruker.south west) -- (pameldt-til.north east);

  % Brukerprofil -- har_økt -- Økt
  \draw[->] (bruker-okt.west) -- (bruker.east);
  \draw[double, double distance=1.5pt] (bruker-okt.east) -- (okt.west);

  % Brukerprofil -- har_prikk -- Prikk
  \draw[->] (bruker-prikk.north west) -- (bruker.south east);
  \draw[double, double distance=1.5pt] (bruker-prikk.south east) -- (prikk.north west);

\end{tikzpicture}
\end{document}
