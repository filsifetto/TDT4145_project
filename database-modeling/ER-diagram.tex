% Kompiler med: pdflatex ER-diagram.tex
% Rekompiler alltid etter endringer i diagrammet.
%
% ER-diagram med tikz-er2: entity, weak entity, relationship, ident relationship,
% \key{} for nøkler, \dashuline{} for partiell nøkkel (ulem, lastet av tikz-er2).
% Entitetsnavn: \entnavn{...} (fet, oppreist) for å skille fra attributter (kursiv).
\documentclass[border=10pt]{standalone}
\usepackage{xcolor}
\usepackage{tikz-er2}
\usetikzlibrary{positioning}

\definecolor{entityblue}{RGB}{173, 216, 230}
\newcommand{\entnavn}[1]{\normalfont\textbf{#1}}

\begin{document}
\begin{tikzpicture}[
  every entity/.style={
    fill=entityblue,
    minimum width=2.6cm,
    minimum height=2.2cm,
    align=left,
    font=\itshape,
    inner sep=4pt,
  },
  every weak entity/.style={
    fill=entityblue,
    minimum width=2.6cm,
    minimum height=2.2cm,
    align=left,
    font=\itshape,
    inner sep=4pt,
  },
  every relationship/.style={
    fill=entityblue,
    aspect=2,
    minimum width=1.5cm,
    minimum height=1cm,
    inner sep=2pt,
  },
  every ident relationship/.style={
    fill=entityblue,
    aspect=2,
    minimum width=1.5cm,
    minimum height=1cm,
    inner sep=2pt,
  }
]

  % === VENSTRE KOLONNE: Senter-stammen + Gruppeaktivitet ===
  \node[entity] (senter) at (0, 10) {\entnavn{Senter}\\
    \key{ID}\\
    navn\\
    adresse\\
    \hspace{1em} gate\\
    \hspace{1em} nummer\\
    åpningstid\\
    stengetid\\
    bemannet\_fra\\
    bemannet\_til};
  \node[ident relationship] (senter-sal) at (0, 7) {\normalsize har\_sal};
  \node[weak entity] (sal) at (0, 4) {\entnavn{Sal}\\
    \dashuline{ID}\\
    type\\
    kapasitet};
  \node[ident relationship] (sal-tredemolle) at (-3.5, 4) {\normalsize har\_tredemølle};
  \node[weak entity] (tredemolle) at (-3.5, 1) {\entnavn{Tredemølle}\\
    \dashuline{nummer}\\
    produsent\\
    maksimal\_hastighet\\
    maksimal\_stigning};
  \node[ident relationship] (sal-sykkel) at (0, 1) {\normalsize har\_sykkel};
  \node[weak entity] (sykkel) at (0, -2) {\entnavn{Sykkel}\\
    \dashuline{nummer}\\
    bluetooth};
  \node[entity] (gruppe) at (0, -5) {\entnavn{Gruppeaktivitet}\\
    \key{ID}\\
    type\\
    beskrivelse};

  % === ØVERSTE RAD: Senter -- har -- Fasilitet ===
  \node[relationship] (senter-fasilitet) at (4, 10) {\normalsize har};
  \node[entity] (fasilitet) at (8, 10) {\entnavn{Fasilitet}\\
    \key{ID}\\
    type\\
    beskrivelse};

  % === MIDTRAD: Gruppeaktivitet -- ansvarlig_for -- Brukerprofil -- har_økt -- Økt ===
  \node[relationship] (ansvarlig-for) at (4, -5) {\normalsize ansvarlig\_for};
  \node[entity] (bruker) at (8, -2) {\entnavn{Brukerprofil}\\
    \key{ID}\\
    navn\\
    epost\\
    telefon};
  \node[ident relationship] (bruker-okt) at (11, -2) {\normalsize har\_økt};
  \node[weak entity] (okt) at (14, -2) {\entnavn{Økt}\\
    \dashuline{ID}\\
    innsjekk\\
    utsjekk};

  % === HØYRE KOLONNE: er_medlem, Idrettslag, har_prikk, Prikk ===
  \node[relationship] (er-medlem) at (12, 1) {\normalsize er\_medlem};
  \node[entity] (idrett) at (18, 4) {\entnavn{Idrettslag}\\
    \key{ID}\\
    navn\\
    beskrivelse};
  \node[ident relationship] (bruker-prikk) at (18, -5) {\normalsize har\_prikk};
  \node[weak entity] (prikk) at (18, -8) {\entnavn{Prikk}\\
    \dashuline{dato\_fikk\_prikk}};

  % Kantene med pilnotasjon for kardinalitet (pil fra relasjon mot «én»-siden)
  \draw (senter.east) -- (senter-fasilitet.west);
  \draw (fasilitet.west) -- (senter-fasilitet.east);
  \draw (senter.south) -- (senter-sal.north);
  \draw[double, double distance=1.5pt] (senter-sal.south) -> (sal.north);
  \draw (sal.west) -- (sal-tredemolle.east);
  \draw[double, double distance=1.5pt] (sal-tredemolle.south) -> (tredemolle.north);
  \draw (sal.south) -- (sal-sykkel.north);
  \draw[double, double distance=1.5pt] (sal-sykkel.south) -> (sykkel.north);
  \draw (ansvarlig-for.west) -> (gruppe.east);
  \draw (ansvarlig-for.east) -- (bruker.west);
  \draw (bruker.east) -- (bruker-okt.west);
  \draw[double, double distance=1.5pt] (bruker-okt.east) -> (okt.west);
  \draw (bruker.south east) -- (bruker-prikk.north west);
  \draw[double, double distance=1.5pt] (bruker-prikk.south east) -> (prikk.north west);
  \draw (bruker.north east) -- (er-medlem.south west);
  \draw (idrett.south west) -- (er-medlem.north east);

\end{tikzpicture}
\end{document}
